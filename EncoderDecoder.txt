import base64
import math

# --- Configuration ---
input_file = "version3.zip"
base_output_name = "version3" # Will become version3-1.txt, version3-2.txt, etc.
num_files = 4 # Change this parameter to the number of files you want

try:
    # 1. Read the document in binary mode
    with open(input_file, "rb") as doc_file:
        binary_data = doc_file.read()
        
    # 2. Convert binary data to a Base64 string
    base64_encoded = base64.b64encode(binary_data).decode('utf-8')
    total_length = len(base64_encoded)
    
    # 3. Calculate exactly how many characters go into each file
    # We use math.ceil to ensure any remainder gets pushed into the final file
    chunk_size = math.ceil(total_length / num_files)
    
    # 4. Slice the string and save to multiple files
    for i in range(num_files):
        # Calculate the start and end positions for the slice
        start_index = i * chunk_size
        end_index = min((i + 1) * chunk_size, total_length) 
        
        # Extract the chunk
        chunk = base64_encoded[start_index:end_index]
        
        # Create the file name (e.g., version3-1.txt)
        output_txt = f"{base_output_name}-{i+1}.txt"
        
        # Write the chunk to the file
        with open(output_txt, "w") as text_file:
            text_file.write(chunk)
            
        print(f"Saved: {output_txt} (Characters: {len(chunk)})")
        
    print("\nSuccess! Splitting complete.")

except FileNotFoundError:
    print(f"Error: Could not find {input_file}. Make sure the file is in the same folder.")
	


import base64

# --- Configuration ---
base_input_name = "version3" # The prefix of your text files
num_files = 4 # Must match the number of files you split it into
output_file = "restored_version3.zip" # The final combined file

combined_base64 = ""

try:
    # 1. Read all the text files in order and combine their strings
    for i in range(num_files):
        input_txt = f"{base_input_name}-{i+1}.txt"
        
        with open(input_txt, "r") as text_file:
            # Append the contents of this file to our master string
            combined_base64 += text_file.read()
            
        print(f"Read and merged: {input_txt}")

    # 2. Convert the massive Base64 string back to binary data
    binary_data = base64.b64decode(combined_base64)
    
    # 3. Save the binary data back to the original format
    with open(output_file, "wb") as doc_file:
        doc_file.write(binary_data)
        
    print(f"\nSuccess! The original file has been restored as: {output_file}")

except FileNotFoundError as e:
    print(f"Error: Could not find one of the text chunks. {e}")
except base64.binascii.Error:
    print("Error: The combined Base64 string is invalid. Make sure all chunks are present and in order.")
